### 基于栈结构 进行十进制转二进制

```javascript
function dec2bin(num) {
      //  定义栈对象
      let stack = new Stack()
      // 循环
      while(num > 0) {
        //  获取余数，并放入栈中
        stack.push(num % 2)
        //  将整除后的整数（向下取整）作为下次循环的数字
        num = Math.floor(num / 2)
      }
      //  从栈中取出0和1 
      let binaryString = ''
      while(!stack.isEmpty()) {
        binaryString += stack.pop()
      }
      return binaryString
    }
    console.log('dec2bin', dec2bin(10))
    console.log('dec2bin', dec2bin(100))
    console.log('dec2bin', dec2bin(2))
```
